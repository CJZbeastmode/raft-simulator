## Step 2 — Update `AppendEntries` handler (30 min)

This is the core work. Replace your current handler with this, and **read every comment**:

```go
func (rf *Raft) AppendEntries(args *AppendEntriesArgs, reply *AppendEntriesReply) {
    rf.mu.Lock()
    defer rf.mu.Unlock()

    reply.Term = rf.currentTerm
    reply.Success = false

    // Rule 1: stale leader → reject
    if args.Term < rf.currentTerm {
        return
    }

    // Valid message from current leader → reset timer, update term
    rf.lastHeartbeat = time.Now()
    rf.state = "follower"
    rf.currentTerm = args.Term
    reply.Term = rf.currentTerm

    // Rule 2: our log doesn't have an entry at PrevLogIndex → reject
    if args.PrevLogIndex >= len(rf.log) {
        return
    }

    // Rule 3: entry at PrevLogIndex has wrong term → reject
    if rf.log[args.PrevLogIndex].Term != args.PrevLogTerm {
        return
    }

    // Happy path: log is consistent up to PrevLogIndex
    // Append any new entries, overwriting conflicts
    for i, entry := range args.Entries {
        idx := args.PrevLogIndex + 1 + i
        if idx < len(rf.log) {
            // Overwrite only if there's a term conflict (Rule 3 of Figure 2)
            if rf.log[idx].Term != entry.Term {
                rf.log = rf.log[:idx] // truncate from conflict point
                rf.log = append(rf.log, args.Entries[i:]...)
                break
            }
            // Identical entry already exists — skip
        } else {
            // Beyond current log end — just append the rest
            rf.log = append(rf.log, args.Entries[i:]...)
            break
        }
    }

    // Advance commitIndex if leader says so
    if args.LeaderCommit > rf.commitIndex {
        lastNewIndex := args.PrevLogIndex + len(args.Entries)
        if args.LeaderCommit < lastNewIndex {
            rf.commitIndex = args.LeaderCommit
        } else {
            rf.commitIndex = lastNewIndex
        }
    }

    reply.Success = true
}
```

The key insight: `PrevLogIndex`/`PrevLogTerm` is the **consistency check**. The leader says "you must have this entry before I send you new ones." If you don't, reject — the leader will back up and retry.

```go
func (rf *Raft) sendHeartbeats() {
    rf.mu.Lock()
    term := rf.currentTerm
    leaderId := rf.me
    commitIndex := rf.commitIndex
    rf.mu.Unlock()

    for i := range rf.peers {
        if i == rf.me {
            continue
        }

        go func(peer int) {
            rf.mu.Lock()
            // Still leader?
            if rf.state != "leader" {
                rf.mu.Unlock()
                return
            }
            ni := rf.nextIndex[peer]
            prevLogIndex := ni - 1
            prevLogTerm := rf.log[prevLogIndex].Term
            entries := make([]LogEntry, len(rf.log[ni:]))
            copy(entries, rf.log[ni:])
            rf.mu.Unlock()

            args := &AppendEntriesArgs{
                Term:         term,
                LeaderId:     leaderId,
                PrevLogIndex: prevLogIndex,
                PrevLogTerm:  prevLogTerm,
                Entries:      entries,
                LeaderCommit: commitIndex,
            }
            reply := &AppendEntriesReply{}

            ok := rf.peers[peer].Call("Raft.AppendEntries", args, reply)
            if !ok {
                return
            }

            rf.mu.Lock()
            defer rf.mu.Unlock()

            // Stale reply — ignore
            if rf.state != "leader" || rf.currentTerm != term {
                return
            }

            // Demote if we see higher term
            if reply.Term > rf.currentTerm {
                rf.currentTerm = reply.Term
                rf.state = "follower"
                rf.votedFor = -1
                return
            }

            if reply.Success {
                // Update nextIndex and matchIndex
                newMatch := prevLogIndex + len(entries)
                if newMatch > rf.matchIndex[peer] {
                    rf.matchIndex[peer] = newMatch
                }
                rf.nextIndex[peer] = rf.matchIndex[peer] + 1
            } else {
                // Follower rejected — back up nextIndex and retry next heartbeat
                // (naive backtracking; Session 6 adds acceleration)
                if rf.nextIndex[peer] > 1 {
                    rf.nextIndex[peer]--
                }
            }
        }(i)
    }
}
```
